1. Customer Support Widget

<!-- MyWaterCloset Support Chat --> <script src="https://hastets.b-cdn.net/widget/widget.iife.js" data-workspace-id="mywatercloset" data-api-url="https://api-production-de24c.up.railway.app" data-ws-host="api-production-de24c.up.railway.app" data-ws-key="reusable-chat-key" data-position="bottom-right" data-title="Need Help?" data-subtitle="We're here to help!" data-welcome-message="Welcome to MyWaterCloset! How can we help you today?" ></script>

2. Authenticated User Chat (Buyer-Seller)

<!-- MyWaterCloset Messaging (requires backend session) --> <script src="https://hastets.b-cdn.net/widget/widget.iife.js" data-workspace-id="mywatercloset" data-api-url="https://api-production-de24c.up.railway.app" data-ws-host="api-production-de24c.up.railway.app" data-ws-key="reusable-chat-key" data-position="bottom-left" data-title="Messages" data-user-token="{{ $chatSession['token'] }}" data-user-id="{{ auth()->id() }}" ></script>

3. Web Component Syntax

<!-- Using Web Component --> <script src="https://hastets.b-cdn.net/widget/widget.iife.js"></script> <chat-widget workspace-id="mywatercloset" api-url="https://api-production-de24c.up.railway.app" ws-host="api-production-de24c.up.railway.app" ws-key="reusable-chat-key" position="bottom-right" title="MyWaterCloset Support" ></chat-widget>

4. JavaScript API

// Programmatic widget control <script src="https://hastets.b-cdn.net/widget/widget.iife.js"></script> <script> const widget = new window.ChatWidget({ workspaceId: 'mywatercloset', apiUrl: 'https://api-production-de24c.up.railway.app', wsHost: 'api-production-de24c.up.railway.app', wsKey: 'reusable-chat-key', position: 'bottom-right', title: 'Chat Support', }); // Initialize when ready widget.init(); // Open chat programmatically document.getElementById('contact-btn').onclick = () => { window.chatWidget.open(); }; </script>
